function makeCall() {
    phone.dial({
    destination: phone.cleanPhoneNumber(callToInput.value),
    mediaType: 'audio',
    localMedia: document.getElementById('local'),
    remoteMedia: document.getElementById('remote')

}

function answerCall() {
    phone.answer({

    localMedia: document.getElementById('local'),
    remoteMedia: document.getElementById('remote')
    });
}
